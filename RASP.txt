import RPi.GPIO as GPIO, time
GPIO.setmode(GPIO.BCM)
GPIO.setup(17, GPIO.IN)
try:
    while True:
        print("Obstacle detected" if GPIO.input(17) == 0 else "No Obstacle")
        time.sleep(0.5)
except KeyboardInterrupt:
    GPIO.cleanup()
